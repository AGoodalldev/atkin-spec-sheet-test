        /** ================== NORMALIZATION ================== **/

        const normHeader = (s) => String(s ?? "").replace(/\s*\/\s*/g, " / ").replace(/\s+/g, " ").trim();


        function coerceList(v) {

            if (!v) return [];

            if (Array.isArray(v)) return v.map(String).map((s) => s.trim()).filter(Boolean);

            if (typeof v === "object") {

                const o = v;

                if (Array.isArray(o.values)) return o.values.map(String).map((s) => s.trim()).filter(Boolean);

                if (Array.isArray(o.items)) return o.items.map(String).map((s) => s.trim()).filter(Boolean);

            }

            if (typeof v === "string") {

                return v.split(/[\n\r,;|]+/).map((s) => s.trim()).filter(Boolean);

            }

            return [];

        }


        function normalizeOptions(raw) {

            try {

                const body = raw && typeof raw === "object" ? (raw.data ?? raw) : {};

                const out = {};

                for (const [k, v] of Object.entries(body || {})) {

                    const nk = normHeader(k);

                    const arr = coerceList(v);

                    if (nk && arr.length) out[nk] = Array.from(new Set(arr));

                }

                return out;

            } catch {

                return {};

            }

        }


        function normalizeModels(raw) {

            const candidate = raw && typeof raw === "object" ? (raw.models ?? raw.data ?? raw) : {};

            if (!candidate || typeof candidate !== "object" || Array.isArray(candidate)) return {};


            const out = {};

            for (const [modelName, fields] of Object.entries(candidate)) {

                const rec = fields && typeof fields === "object" ? fields : {};

                const normalized = {};

                for (const [fk, fv] of Object.entries(rec)) {

                    normalized[normHeader(fk)] = String(fv ?? "");

                }

                out[String(modelName)] = normalized;

            }

            return out;

        }

                        next.delete(field);

                    }
